import { getCustomStaticPath } from '@/utils/getCustomStaticPath';

export const meta = {
  title: 'Enable and disable analytics',
  description: 'Learn how to enable/disable analytics using Amplify.',
  platforms: ['javascript', 'react-native', 'swift', 'android']
};

export const getStaticPaths = async () => {
  return getCustomStaticPath(meta.platforms);
};

export function getStaticProps(context) {
  return {
    props: {
      platform: context.params.platform,
      meta
    }
  };
}

## Disable Analytics

Analytics is enabled by default when you configure it in your app. To disable Analytics in your app use the `disable` function:

import android0 from '/src/fragments/lib/analytics/android/enable-disable/disable.mdx';

<Fragments fragments={{ android: android0 }} />

import flutter0 from '/src/fragments/lib/analytics/flutter/enable-disable/disable.mdx';

<Fragments fragments={{ flutter: flutter0 }} />

import ios0 from '/src/fragments/lib/analytics/ios/enable-disable/disable.mdx';

<Fragments fragments={{ swift: ios0 }} />

import js0 from '/src/fragments/lib/analytics/js/enable-disable/disable.mdx';

<Fragments fragments={{ javascript: js0 }} />

import reactnative0 from '/src/fragments/lib/analytics/js/enable-disable/disable.mdx';

<Fragments fragments={{ 'react-native': reactnative0 }} />

## Enable Analytics

To enable analytics you can use the `enable` function in your app:

import android1 from '/src/fragments/lib/analytics/android/enable-disable/enable.mdx';

<Fragments fragments={{ android: android1 }} />

import flutter1 from '/src/fragments/lib/analytics/flutter/enable-disable/enable.mdx';

<Fragments fragments={{ flutter: flutter1 }} />

import ios1 from '/src/fragments/lib/analytics/ios/enable-disable/enable.mdx';

<Fragments fragments={{ swift: ios1 }} />

import js1 from '/src/fragments/lib/analytics/js/enable-disable/enable.mdx';

<Fragments fragments={{ javascript: js1 }} />

import reactnative1 from '/src/fragments/lib/analytics/js/enable-disable/enable.mdx';

<Fragments fragments={{ 'react-native': reactnative1 }} />
