## Disable Analytics

Analytics is enabled by default when you configure it in your app. To disable Analytics in your app use the `disable` function:

import js0 from '/src/fragments/lib/analytics/js/enable-disable/disable.mdx';

<Fragments
  fragments={{
    javascript: js0,
    angular: js0,
    nextjs: js0,
    react: js0,
    vue: js0
  }}
/>

import reactNative0 from '/src/fragments/lib/analytics/js/enable-disable/disable.mdx';

<Fragments fragments={{ 'react-native': reactNative0 }} />

import android0 from '/src/fragments/lib/analytics/android/enable-disable/disable.mdx';

<Fragments fragments={{ android: android0 }} />

import flutter0 from '/src/fragments/lib/analytics/flutter/enable-disable/disable.mdx';

<Fragments fragments={{ flutter: flutter0 }} />

import ios0 from '/src/fragments/lib/analytics/ios/enable-disable/disable.mdx';
