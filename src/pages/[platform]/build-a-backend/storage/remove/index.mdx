Delete stored data from the storage bucket. To remove a file, you need to provide the `path`. 

Files can only be deleted for the current user. Deletion capabilities do not extend to files associated with other users.

<BlockSwitcher>
<Block name="With Path (Above v6.x.x)">
```javascript
import { remove } from 'aws-amplify/storage';

try {
  await remove({ 
    path: 'album/2024/1.jpg',
    // Alternatively, path: ({identityId}) => `album/{identityId}/1.jpg`
  });
} catch (error) {
  console.log('Error ', error);
}
```
</Block>
<Block name="With Key (Deprecated)">
```javascript
import { remove } from 'aws-amplify/storage';

try {
  await remove({ 
    key: "abc.txt", 
    options: { 
      accessLevel: 'protected' // defaults to `guest` but can be 'private' | 'protected' | 'guest'
    } 
  });
} catch (error) {
  console.log('Error ', error);
}
```
</Block>
</BlockSwitcher>
