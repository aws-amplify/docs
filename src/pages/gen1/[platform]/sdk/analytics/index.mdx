import { getCustomStaticPath } from '@/utils/getCustomStaticPath';
import { getChildPageNodes } from '@/utils/getChildPageNodes';

export const meta = {

  description: 'Make informed decisions with drop-in analytics to track user sessions, custom user attributes and in-app metrics',
  platforms: [

    'swift'
  ],
  route: '/gen1/[platform]/sdk/analytics',
  canonicalObjects: [

      platforms: [
        'swift',

      canonicalPath: '/gen1/swift/sdk/analytics/'

};

export const getStaticPaths = async () => {

};

  return {
    props: {
      platform: context.params.platform,

      childPageNodes
    }
  };
}
