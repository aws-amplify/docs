You can record receipt of a push notification with Amazon Pinpoint by calling `Amplify.Notifications.Push.recordNotificationReceived(userInfo)` from the `application(_:didReceiveRemoteNotification:)` delegate method of your Application Delegate.

<BlockSwitcher>
<Block name="iOS">

```swift
// Called when a notification is received
//
// Requires background processing->remote notifications entitlement
// and "content-available" : 1 in the notification
//
func application(
    _ application: UIApplication,
    didReceiveRemoteNotification userInfo: [AnyHashable: Any]
) async -> UIBackgroundFetchResult {
    
    do {
        try await Amplify.Notifications.Push.recordNotificationReceived(userInfo)
    } catch {
        print("Error recording receipt of notification: \(error)")
    }
    
    return .newData
}
```

</Block>

```swift

func application(

    didReceiveRemoteNotification userInfo: [String : Any]

    }
}
```
