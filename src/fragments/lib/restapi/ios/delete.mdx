## Delete data

```swift
func deleteTodo() async {
    let request = RESTRequest(path: "/todo")
    do {
        let data = try await Amplify.API.delete(request: request)
        let str = String(decoding: data, as: UTF8.self)
        print("Success: \(str)")
    } catch(let error) {
        print("Failed: ", error)
    }
}
```
