func postTodo() async {
    let message = #"{"message": "my new Todo"}"#
    let request = RESTRequest(path: "/todo", body: message.data(using: .utf8))
    do {
        let data = try await Amplify.API.post(request: request)
        let str = String(decoding: data, as: UTF8.self)
        print("Success: \(str)")
    } catch(let error) {
        print("Failed", error)
    }
}
