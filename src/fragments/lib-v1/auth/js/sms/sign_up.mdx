import { signUp } from 'aws-amplify/auth';

async function handleSignUp(
	username: string,
	password: string,
	phone_number: string,
	email: string
) {
	try {
		await signUp({
			username,
			password,
			options: {
				userAttributes: {
					phone_number,
					email,
				},
			},
		});
	} catch (error) {
		console.log(error);
	}
}
